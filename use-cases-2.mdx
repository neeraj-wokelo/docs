### Full two-step workflow — list products then fetch reviews

<CodeGroup>

```bash cURL — Step 1: list products
curl --location 'https://api.akta.pro/api/v1/company/product-reviews?company=tesla.com' \
  --header 'x-api-key: <YOUR_API_KEY>'
```

```python Python
import requests

HEADERS = {"x-api-key": "<YOUR_API_KEY>"}

# ── Step 1: Get the product catalog ──────────────────────────
list_response = requests.get(
    "https://api.akta.pro/api/v1/company/product-reviews",
    headers=HEADERS,
    params={"company": "tesla.com"}
)

products = list_response.json()["data"]["product_reviews"]
print(f"Products found: {len(products)}")
for p in products:
    print(f"  {p['id']:>10}  {p['product_name']}  —  {p['main_category']}")

# ── Step 2: Fetch reviews for selected product IDs ───────────
product_ids = [p["id"] for p in products[:2]]   # take first two

review_response = requests.get(
    "https://api.akta.pro/api/v1/company/product-reviews",
    headers=HEADERS,
    params=[("company", "tesla.com")]
           + [("products", pid) for pid in product_ids]
           + [("limit", 10), ("offset", 0)]
)

data = review_response.json()
print(f"\nCredits consumed: {data['credits_consumed']}")

for product in data["data"]["product_reviews"]:
    if "reviews" not in product:
        continue
    rating = product["rating"].get("g2")
    total  = product["reviews"]["total"]
    dist   = product["star_distribution"].get("g2", {})
    total_dist = sum(dist.values())

    print(f"\n{'='*55}")
    print(f"  {product['product_name']}")
    print(f"  Rating: {rating}/5.0  |  {total} reviews total")

    if total_dist > 0:
        for star in sorted(dist.keys(), key=int, reverse=True):
            pct = round(dist[star] / total_dist * 100)
            bar = "█" * (pct // 5)
            print(f"  {star}★  {bar:<20}  {pct}%  ({dist[star]})")

    print(f"\n  Most recent reviews:")
    for review in product["reviews"]["data"][:3]:
        print(f"  [{review['review_rating']}★]  {review['review_title']}")
        print(f"    {review['publish_date'][:10]}  |  {review['reviewer_company_size']}")
        print(f"    {review['review_content'][:150].strip()}...")
```

### Paginating through reviews for a high-volume product

```python
import requests

HEADERS = {"x-api-key": "<YOUR_API_KEY>"}
PRODUCT_ID  = "177388"
COMPANY     = "tesla.com"
PAGE_SIZE   = 50

all_reviews = []
offset      = 0

while True:
    r = requests.get(
        "https://api.akta.pro/api/v1/company/product-reviews",
        headers=HEADERS,
        params=[
            ("company",  COMPANY),
            ("products", PRODUCT_ID),
            ("limit",    PAGE_SIZE),
            ("offset",   offset)
        ]
    )
    data    = r.json()
    product = data["data"]["product_reviews"][0]
    batch   = product["reviews"]["data"]

    if not batch:
        break

    all_reviews.extend(batch)
    total = product["reviews"]["total"]
    print(f"Fetched {len(all_reviews)} / {total} reviews...")

    if len(all_reviews) >= total:
        break

    offset += PAGE_SIZE

print(f"\nTotal reviews retrieved: {len(all_reviews)}")
```

### Using `review_question_answers` for structured processing

Use `review_question_answers` for clean structured extraction rather than parsing the concatenated `review_content` string.

```python
for review in product["reviews"]["data"]:
```

### JavaScript / Node.js

```javascript
const fetch = require("node-fetch");

const API_KEY = process.env.AKTA_API_KEY;
const BASE    = "https://api.akta.pro/api/v1/company/product-reviews";

async function getProductList(company) {
  const res = await fetch(`${BASE}?company=${company}`, {
    headers: { "x-api-key": API_KEY }
  });
  if (!res.ok) throw new Error(`API error: ${res.status}`);
  return (await res.json()).data.product_reviews;
}

async function getProductReviews(company, productIds, limit = 10, offset = 0) {
  const params = new URLSearchParams({ company, limit, offset });
  productIds.forEach(id => params.append("products", id));

  const res = await fetch(`${BASE}?${params}`, {
    headers: { "x-api-key": API_KEY }
  });
  if (!res.ok) throw new Error(`API error: ${res.status}`);
  return (await res.json());
}

// Step 1: list products
getProductList("tesla.com").then(products => {
  console.log(`Products: ${products.length}`);
  products.forEach(p => console.log(`  ${p.id}  ${p.product_name}`));

  // Step 2: fetch reviews for first product
  const ids = products.slice(0, 1).map(p => p.id);
  return getProductReviews("tesla.com", ids, 5);

}).then(data => {
  const product = data.data.product_reviews[0];
  const rating  = product.rating?.g2;
  const total   = product.reviews?.total;
  console.log(`\n${product.product_name}: ${rating}/5.0 (${total} reviews)`);
  product.reviews.data.forEach(r => {
    console.log(`  [${r.review_rating}★] ${r.review_title}`);
  });
});
```
