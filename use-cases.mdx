### Company news — positive funding and deal activity

Fetch all positive news about a company over a date range, filtered to funding and M&A event types.

<CodeGroup>

```bash cURL
curl --location 'https://api.akta.pro/api/v1/news?company=canva.com&sentiment=positive&start_date=2024-01-01&limit=25' \
  --header 'x-api-key: <YOUR_API_KEY>'
```

```python Python
import requests

HEADERS = {"x-api-key": "<YOUR_API_KEY>"}

response = requests.get(
    "https://api.akta.pro/api/v1/news",
    headers=HEADERS,
    params={
        "company":    "canva.com",
        "sentiment":  "positive",
        "start_date": "2024-01-01",
        "limit":      25
    }
)

data     = response.json()
articles = data["data"]

print(f"Found {data['total']} articles (showing {len(articles)})\n")

for article in articles:
    primary_tag = next(
        (t["name"] for t in article.get("tags", []) if t.get("is_primary")),
        "—"
    )
    print(f"[{article['published_date']}] [{primary_tag}]")
    print(f"  {article['title']}")
    print(f"  {article['publisher']} | Score: {article['news_score']:.2f}")
    print(f"  {article['ai_summary'][:180]}...")
    print()
```

### Industry news monitoring

Monitor all news across an industry, sorted by relevance score.

```python
import requests

HEADERS = {"x-api-key": "<YOUR_API_KEY>"}

response = requests.get(
    "https://api.akta.pro/api/v1/news",
    headers=HEADERS,
    params={
        "industry":   "Fintech",
        "start_date": "2024-06-01",
        "limit":      50
    }
)

articles = response.json()["data"]

# Sort by news_score descending for triage
articles.sort(key=lambda x: x.get("news_score", 0), reverse=True)

print(f"Top articles by relevance score:\n")
for article in articles[:10]:
    print(f"  [{article['news_score']:.2f}] {article['title']}")
    print(f"    {article['publisher']} | {article['published_date']} | {article['sentiment']}")
```

### Adverse signal monitoring — negative news for a watchlist

Pull negative sentiment news across a set of companies to surface early warning signals.

```python
import requests, concurrent.futures

HEADERS   = {"x-api-key": "<YOUR_API_KEY>"}
WATCHLIST = ["stripe.com", "brex.com", "ramp.com", "mercury.com"]

def get_negative_news(domain):
    r = requests.get(
        "https://api.akta.pro/api/v1/news",
        headers=HEADERS,
        params={
            "company":    domain,
            "sentiment":  "negative",
            "start_date": "2024-01-01",
            "limit":      10
        }
    )
    articles = r.json().get("data", [])
    return domain, articles

# Fetch all in parallel
with concurrent.futures.ThreadPoolExecutor(max_workers=4) as pool:
    results = dict(pool.map(lambda d: get_negative_news(d), WATCHLIST))

for domain, articles in results.items():
    if not articles:
        continue
    print(f"\n{domain} — {len(articles)} adverse signals")
    for article in articles[:3]:
        primary_tag = next(
            (t["name"] for t in article.get("tags", []) if t.get("is_primary")), "—"
        )
        print(f"  [{article['published_date']}] [{primary_tag}] {article['title']}")
        print(f"    Score: {article['news_score']:.2f} | {article['ai_summary'][:120]}...")
```

### Extracting full article text for LLM processing

Use `full_text` to feed complete article content into a summarisation or classification pipeline.

```python
import requests

HEADERS = {"x-api-key": "<YOUR_API_KEY>"}

response = requests.get(
    "https://api.akta.pro/api/v1/news",
    headers=HEADERS,
    params={
        "company": "openai.com",
        "type":    "Equity Fund-Raising",
        "limit":   20
    }
)

articles = response.json()["data"]

# Filter to articles with full text available
with_text = [a for a in articles if a.get("full_text", "").strip()]
print(f"{len(with_text)} of {len(articles)} articles have full text\n")

for article in with_text[:3]:
    word_count = len(article["full_text"].split())
    print(f"Title:     {article['title']}")
    print(f"Publisher: {article['publisher']} | Words: {word_count}")
    print(f"Summary:   {article['ai_summary']}")
    print()
    # Pass article["full_text"] to your LLM pipeline here
```

### Paginating through large result sets

When `total` exceeds `limit`, paginate using `offset`.

```python
import requests

HEADERS = {"x-api-key": "<YOUR_API_KEY>"}
PARAMS  = {
    "company":    "apple.com",
    "start_date": "2024-01-01",
    "limit":      100,
    "offset":     0
}

all_articles = []

while True:
    response = requests.get(
        "https://api.akta.pro/api/v1/news",
        headers=HEADERS,
        params=PARAMS
    )
    data  = response.json()
    batch = data.get("data", [])
    all_articles.extend(batch)

    print(f"Fetched {len(all_articles)} / {data['total']} articles")

    if len(all_articles) >= data["total"] or not batch:
        break

    PARAMS["offset"] += PARAMS["limit"]

print(f"\nTotal retrieved: {len(all_articles)}")
```

### Resolving company mentions across an industry feed

Use `company_names` to map all entities referenced in articles without parsing article text.

```python
import requests
from collections import Counter

HEADERS = {"x-api-key": "<YOUR_API_KEY>"}

response = requests.get(
    "https://api.akta.pro/api/v1/news",
    headers=HEADERS,
    params={
        "industry":   "Artificial Intelligence",
        "start_date": "2024-09-01",
        "limit":      100
    }
)

articles = response.json()["data"]

# Count non-primary company mentions (i.e. companies mentioned in context)
mention_counter = Counter()
for article in articles:
    for co in article.get("company_names", []):
        if not co.get("is_primary"):
            mention_counter[co["name"]] += 1

print("Most-mentioned companies in AI news (non-primary):")
for name, count in mention_counter.most_common(10):
    print(f"  {name}: {count} mentions")
```

### JavaScript / Node.js

```javascript
const fetch = require("node-fetch");

async function getNews(params = {}) {
  const query = new URLSearchParams({
    limit:  params.limit  || 10,
    offset: params.offset || 0,
    ...(params.company   && { company:   params.company }),
    ...(params.industry  && { industry:  params.industry }),
    ...(params.sentiment && { sentiment: params.sentiment }),
    ...(params.startDate && { start_date: params.startDate }),
    ...(params.endDate   && { end_date:   params.endDate }),
    ...(params.type      && { type:       params.type })
  });

  const response = await fetch(
    `https://api.akta.pro/api/v1/news?${query}`,
    {
      headers: {
        "x-api-key":    process.env.AKTA_API_KEY,
        "Content-Type": "application/json"
      }
    }
  );

  if (!response.ok) {
    throw new Error(`API error: ${response.status} ${response.statusText}`);
  }

  return response.json();
}

// Example: positive funding news for Canva
getNews({
  company:   "canva.com",
  sentiment: "positive",
  startDate: "2024-01-01",
  limit:     10
}).then(data => {
  console.log(`Found ${data.total} articles`);
  data.data.forEach(a => {
    console.log(`[${a.news_score.toFixed(2)}] ${a.title}`);
    console.log(`  ${a.publisher} | ${a.published_date}`);
  });
});
```
